% --- COMPACT TITLE CONFIGURATION ---
\usepackage{titling}
\setlength{\droptitle}{-4em} % Pulls the title block up

\posttitle{\par\end{center}\vspace{-1.0em}}

% --- SECTION CONFIGURATION ---
\usepackage{titlesec}
\titleformat{\section}{\normalfont\huge\bfseries}{\thesection}{1em}{}[\titlerule]

% --- ZERO-INDENT ABSTRACT COMMAND ---
\newcommand{\tightabstract}[1]{
    \begin{center}
        {\huge\bfseries Abstract}
    \end{center}
    \vspace{0.6em} 
    {
        \setlength{\parindent}{0pt} % Kill indentation for the ABT blocks
        \setlength{\parskip}{0.8em}  % Standard gap between AND/BUT/THEREFORE
        #1
    }
}
